/**
 * 把模板字符串变成tokens数组
 */
export default function parseTemplateToTokens(templateStr){
  var tokens = [];
  var scanner = new Scanner(templateStr)
  let words;
  while(!scanner.eos()){
    words = scanner.scanUtil('{{');
    if(words!=''){
      //尝试判断空格，如果是普通文字空格则去掉，标签中的空格保留
      let isInJJH = false;
      var _words = '';
      for(let i=0;i<words.length;i++){
        //判断是否在标签里
        if(words[i]=='<'){
          isInJJH = true;
        }else if(words[i]=='>'){
          isInJJH = false;
        }
        //不是空格，拼接上
        if(!/\s/.test(words[i])){
          _words += words[i];
        }else{
          //是标签中的空格，拼接上
          if(!isInJJH){
            _words += ' '
          }
        }
      }
      tokens.push(['text',_words])
    }
    scanner.scan('{{')
    words = scanner.scanUtil('}}');
    if(words!=''){
      if(words[0]=='#'){
        tokens.push(['#',words.substring(1)])
      }else if(words[0]=='/'){
        tokens.push(['/',words.substring(1)])
      }else{
        tokens.push(['name',words])
      }
      
    }
    scanner.scan('}}')
  }
  return nestTokens(tokens);
}

/**
 * 扫描器类Scanner.js
 */

export default class Scanner{
  constructor(templateStr){
    this.templateStr = templateStr;
    this.pos = 0;
    //尾巴，开始 是模板字符串原文
    this.tail = templateStr;
  }
//走过指定内容，没有返回值
  scan(tag){
    //跳过tag  {{}}  尾巴去除tag
    if(this.tail.indexOf(tag) ==0 ){
      this.pos += tag.length
      this.tail = this.templateStr.substring(this.pos);
    }
  }
//扫面，直到遇见指定内容，返回之前路过的字符串
  scanUtil(stopTag){
    const pos_backup = this.pos;
    //找到停止符号或者结尾时候停止查找
    while(this.tail.indexOf(stopTag)!=0 && !this.eos()){
      this.pos ++;
      this.tail = this.templateStr.substring(this.pos)
    }
    return this.templateStr.substring(pos_backup,this.pos)
  }
//指针是否到头，返回bool，end of string
  eos(){
    return this.pos >= this.templateStr.length;
  }
}

//nestTokens
/**
 * 折叠tokens，把#和/之间的tokens整合起来，作为下表为3的项
 */
export default function nestTokens(tokens){
    var nestedTokens = [];
    //需要一个栈保存小tokens
    //收集器，指向nestedTokens结果数组
    var collector = nestedTokens;
    var sections = [];
    for(let i=0;i<tokens.length;i++)  {
      let token = tokens[i];
      switch(token[0]){
        case '#':
          collector.push(token)
          //用来收集子元素
          sections.push(token)
          //收集器换人，给token添加下标为2的项，并且收集器指向它
          collector = token[2] = [];
          break;
        case '/':
          var section = sections.pop()
          collector = sections.length >0 ? sections[sections.length-1][2]:nestedTokens
          break;
        default:
          collector.push(token)
      }
    }
    return nestedTokens;
}

import lookup from './lookup';
import parseArray from "./parseArray";
/**
 * 把tokens数组变成dom字符串
 * @param {*} tokens 
 * @param {*} data 
 */
export default function renderTemplate(tokens,data){
  var resultStr = '';
  for(let i=0;i<tokens.length;i++){
    var token = tokens[i];
    if(token[0]=='text'){
      resultStr += token[1];
    }else if(token[0]=='name'){
      //使用lookup，避免a.b.c的情况
      resultStr += lookup(data,token[1])
    }else if(token[0]=='#'){
      resultStr += parseArray(token,data)
    }
  }
  return resultStr;
}

/**lookup
 * 在dataObj中寻找连续点符号的keyName属性
 * {a:{b:{c:10}}}
 * a.b.c === 10
 */

export default function lookup(dataObj,keyName){
  if(keyName.indexOf('.')!=-1 && keyName!='.'){
    let keys = keyName.split('.');
    var temp = dataObj;
    for(let i=0;i<keys.length;i++){
      temp = temp[keys[i]]
    }
    return temp;
  }
  return dataObj[keyName]
};

/**
 * 处理数组，结合renderTemplate实现递归
 * token  ["#", "arr", Array(5)]
 * 调用renderTemplate，次数由data决定
 */

import lookup from "./lookup";
import renderTemplate from "./renderTemplate";

export default function parseArray(token,data){
  //得到整体数据中要使用的部分
  var v = lookup(data,token[1])
  var resultStr = '';
  for(let i=0;i<v.length;i++){
    //这里补充一个。属性，代表当前数据
    resultStr += renderTemplate(token[2],{...v[i],'.':v[i]})
  }
  return resultStr;
}

/**
 * h函数,低配版，必须接受3个参数、
 * h('div',{},'文字')
 * h('div',{},[])

 */

export default function h(sel,data,c){
  if(arguments.length !=3){
    throw new Error('该版本h函数必须传入3个参数')
  }
  if(typeof c =='string' || typeof c == 'number'){
    return vnode(sel,data,undefined,c,undefined)
  }else if(Array.isArray(c)){
    let children = [];
    //数组，里面是h函数，返回值为object
    for(let i=0;i<c.length;i++){
      if(!(typeof c[i] =='object' && c[i].hasOwnProperty('sel'))){
        throw new Error('传入的参数中的项不是h函数')
      }
      children.push(c[i]);
    }
    
    return vnode(sel,data,children,undefined,undefined)
  }else if(typeof c =='object'){
    let children = [c];
    return vnode(sel,data,children,undefined,undefined)
  }else{
    throw new Error('第三个参数类型不对')
  }
}

/**

export default function createElement(vnode){
    let domNode = document.createElement(vnode.sel);
    //判断是否有子节点
    if(vnode.text != '' && (vnode.children == undefined || vnode.children.length == 0)){
        domNode.innerText = vnode.text;
    }else if(Array.isArray(vnode.children) && vnode.children.length >0){
        //内部是节点，循环遍历创建
        for(let i =0;i<vnode.children.length;i++){
            let ch = vnode.children[i];
            //创建dom，并且elm属性指向创建出的dom，但是孤儿节点
            let chDom = createElement(ch);
            //上树
            domNode.appendChild(chDom);
        }
    }
    //补充elm属性
    vnode.elm = domNode;
    //返回elm，elm是一个纯dom对象
    return vnode.elm;
}

import createElement from './createElement';

export default function patchVnode(oldVnode,newVnode){
  //判断新旧节点是不是同一个对象
  if(newVnode == oldVnode){
    return;
  }
  //判断新节点有没有text属性
  if(newVnode.text != null && (newVnode.children ==undefined || newVnode.children.length ==0)){
    //新节点有text属性，新老节点text不同，直接让新节点text写入老的elm中，如果老elm中是children也会消失
    if(newVnode.text != oldVnode.text){
      oldVnode.elm.innerText = newVnode.text
    }
  }else{
    //判断老节点有没有子节点
    if(oldVnode.children != null && oldVnode.children.length >0){
      //新增情况，新增节点插入到未处理节点之前,un指向老节点列表中未处理节点的开头
      updateChildren(oldVnode.elm,oldVnode.children,newVnode.children);
    }else{
      //老的没有children，新的有children
      //清空老节点的内容
      oldVnode.elm.innerText=''
      //遍历新节点的子节点,上树
      for(let i=0;i<newVnode.children.length;i++){
        let dom =  createElement(newVnode.children[i])
        oldVnode.elm.appendChild(dom)
      }
    }
  }
}

export default function updateChildren(parentElm,oldCh,newCh){
  let oldStartIdx = 0,newStartIdx = 0,oldEndIdx=oldCh.length-1,newEndIdx=newCh.length-1;
  let oldStartVnode = oldCh[oldStartIdx],newStartVnode = newCh[newStartIdx];
  let oldEndVnode = oldCh[oldEndIdx],newEndVnode = newCh[newEndIdx];
  let keyMap =null;
  while(oldStartIdx<=oldEndIdx && newStartIdx<=newEndIdx){
    //先略过undefined的值
    if(oldStartVnode == null || oldCh[oldStartIdx] == undefined){
      oldStartVnode = oldCh[++oldStartIdx];
    }else if(oldEndVnode == null || oldCh[oldEndIdx] == undefined){
      oldEndVnode = oldCh[--oldEndIdx];
    }else if(newStartVnode == null || newCh[newStartIdx]==undefined){
      newStartVnode = newCh[++newStartIdx];
    }else if(newEndVnode==null||newCh[newEndIdx]==undefined){
      newEndVnode = newCh[--newEndIdx];

      patchVnode(oldStartVnode,newStartVnode);
      oldStartVnode = oldCh[++oldStartIdx];
      newStartVnode = newCh[++newStartIdx];

          const key=oldCh[i].key;
          if(key != undefined){

export default class Watcher{
  constructor(target,expression,callback){
    console.log('我是warcher构造器');
    this.id = uid++;
    this.target = target;
    this.getter = parsePath(expression);
    this.callback = callback;
    this.value = this.get();
  }
  update(){
    this.run();
  }
  get(){
    //进入依赖收集阶段，让全局的 Dep./ 
    Dep.target = this;
    const obj = this.target;
    var value;
    //只要能找就一直找
    try{
      value = this.getter(obj);
    }finally{
      Dep.target = null;
    }
    return value;
  }
  run(){
    this.getAndInvoke(this.callback);
  }
  getAndInvoke(cb){
    const value = this.get();
    if(value !== this.value || typeof value == 'object'){
      const oldValue = this.value;
      this.value = value;
      cb.call(this.target,value,oldValue);
    }
  }
}

function parsePath(str){
  var segments = str.split('.');
  return (obj)=>{
    for (let i = 0; i < segments.length; i++) {
      if(!obj){
        return
      }
      obj = obj[segments[i]]
    }
    return obj;
  }
}

reactivity ​

Vue 2 中响应式数据是通过 ES5 的 getter/setter 来实现的，当动态向对象上添加 key，或者通过下标的方式修改数组，就不能触发数据变更的检测机制。Proxy 可以为任何一个对象设置一个代理对象，并在代理对象中设置一些自定义的逻辑。

  1. 当一个数据被包装成响应式数据时，它的读取和修改方法会被 Proxy 拦截
  2. 数据被读取时，Proxy 记录 effect 与当前数据的依赖关系
  3. 数据被修改时，触发 effect 运行
  4. 如果调用一下effect(fn)，则之后任何fn()中依赖的数据发生变化，fn()都会被再次调用。这便是 Vue 3 的响应式数据的原理

Vue3中提供了四种创建不同类型的响应式数据的方式：

  • reactive 返回的是一个proxy对象，如果访问返回的proxy对象的属性是一个Object类型，会继续调用reactive，进行深度递归
  • shallowReactive 返回一个proxy对象，但是只有第一层的属性是响应式
  • readonly 返回一个proxy对象，如果属性是对象可以继续调用readonly，但是属性是只读的不可修改，所以在访问阶段也不会进行依赖收集
  • shallowReadonly 返回一个proxy对象，只有第一层的属性是响应式，属性只读不可修改

function createReactiveObject(
  target: Target,
  isReadonly: boolean,
  baseHandlers: ProxyHandler<any>,
  collectionHandlers: ProxyHandler<any>,
  proxyMap: WeakMap<Target, any>
) {
  if (!isObject(target)) {
    if (__DEV__) {
      console.warn(`value cannot be made reactive: ${String(target)}`)
    }
    return target
  }
  // 已经是代理直接返回
  if (
    target[ReactiveFlags.RAW] &&
    !(isReadonly && target[ReactiveFlags.IS_REACTIVE])
  ) {
    return target
  }
  // map中已经存有代理直接返回
  const existingProxy = proxyMap.get(target)
  if (existingProxy) {
    return existingProxy
  }
  // 白名单对象可以被代理响应式
  const targetType = getTargetType(target)
  if (targetType === TargetType.INVALID) {
    return target
  }
  //新建代理对象
  const proxy = new Proxy(
    target,
    //object array===>common	map weakmap set weakset===>collection
    targetType === TargetType.COLLECTION ? collectionHandlers : baseHandlers
  )
  //保存原对象和代理对象的映射
  proxyMap.set(target, proxy)
  return proxy
}

  debuggerEventExtraInfo?: DebuggerEventExtraInfo
) {

export function effect<T = any>(

compiler ​

  1. 调用 baseParse() 方法，解析传入的模板，生成 AST（即 ast 变量）

    代码解析方法，即从源码的第 1 个字符开始逐个字符进行扫描，每当找到可以识别的对象则将该对象放入 AST，并接着之前的源码位置往下继续解析。

  2. 调用 transform() 方法对 AST 进行转换

    针对上一步生成的 AST 进行一些变换，例如针对 v-if 和 v-for 增加一些特殊的节点，以及为了后续渲染性能优化增加一些属性等等

    • 首先使用 tranverseNode() 对 AST 进行遍历，并在每次遍历退出前调用 hook，以便相关逻辑对相应的节点进行修改

    • 如果需要进行静态节点提升，则调用 hoistStatic() 方法提升静态节点

    • 调用 createRootCodegen() 方法生成 codegenNode，以便在后续 generate 阶段使用

  3. 调用 generate() 方法，根据转换后的 AST 生成 render() 方法并返回

    • 生成模块化相关包装代码

    • 根据是否是服务端渲染，应用不同的处理参数生成函数签名

    • 处理 scopeId 和 isSetupInlined 相关逻辑

    • 处理组件、指令等

    • 调用 genNode()，很机械地生成代码，不做详细解析，值得注意的是传入的参数正是 ast.codegenNode，而不是 ast 本身
